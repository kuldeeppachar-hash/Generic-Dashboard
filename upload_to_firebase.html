async function uploadData(){ try{ if(!rawFile){ alert("Select file first"); return; } log("Loading Excel..."); progress(10); const data=await rawFile.arrayBuffer(); const workbook=XLSX.read( data, { type:"array" } ); const sheet= workbook.Sheets[ workbook.SheetNames[0] ]; const rows= XLSX.utils.sheet_to_json( sheet, { defval:"" } ); log("Rows loaded: "+rows.length); progress(20); let stateData={}; let areaData={}; let productData={}; let customerData={}; let dailyData={}; let totalSales=0; let genericRows=0; rows.forEach(r=>{ const division= String( r["Division Name"]||"" ) .trim() .toUpperCase(); if( division!=="GENERIC" ) return; const amount=parseFloat( r["Net Amount"]||0 ); if( isNaN(amount) || amount<=0 ) return; genericRows++; const state= String( r["State"]|| "Unknown" ).trim(); const area= String( r["Sale Offc. Desc."]|| "Unknown" ).trim(); const product= String( r["Product Description"]|| "Unknown" ).trim(); const customer= String( r["Customer Name"]|| "Unknown" ).trim(); let invoiceDate= r["Invoice Date"]; let date="Unknown"; if( typeof invoiceDate==="number" ){ let d=new Date( Math.round( (invoiceDate-25569) * 86400 * 1000 ) ); date= d.toLocaleDateString( "en-IN" ); } else if(invoiceDate){ date= new Date( invoiceDate ) .toLocaleDateString( "en-IN" ); } totalSales+=amount; if(!stateData[state]) stateData[state]={ name:state, amount:0 }; stateData[state] .amount+=amount; if(!areaData[area]) areaData[area]={ name:area, amount:0 }; areaData[area] .amount+=amount; if(!productData[product]) productData[product]={ name:product, amount:0 }; productData[product] .amount+=amount; if(!customerData[customer]) customerData[customer]={ name:customer, amount:0 }; customerData[customer] .amount+=amount; if(!dailyData[date]) dailyData[date]={ date:date, amount:0 }; dailyData[date] .amount+=amount; }); progress(50); log("Generic rows found: "+genericRows); await db.ref( "/state_data" ).set(stateData); await db.ref( "/area_data" ).set(areaData); await db.ref( "/product_data" ).set(productData); await db.ref( "/customer_data" ).set(customerData); await db.ref( "/daily_data" ).set(dailyData); await db.ref( "/meta" ).set({ period_tag: "Generic Division", sap_info: "₹"+ ( totalSales/100000 ).toFixed(2) +" Lacs", updated: new Date() .toLocaleString() }); progress(100); log( "✓ Upload completed successfully" ); alert( "Firebase upload completed" ); } catch(err){ console.error(err); log( "ERROR: "+ err.message ); alert( err.message ); } }
